rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{uid}/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    match /categorizationSessions/{token} {
      allow create: if request.auth != null;
      allow read: if true;
      allow update: if true
        && request.resource.data.userId == resource.data.userId
        && request.resource.data.transactionIds == resource.data.transactionIds;

      match /categories/{catId} {
        allow read: if true;
        allow write: if request.auth != null;
      }

      match /transactions/{txId} {
        allow read: if true;
        allow update: if true
          && request.resource.data.transactionId == resource.data.transactionId
          && request.resource.data.description == resource.data.description
          && request.resource.data.amount == resource.data.amount
          && request.resource.data.date == resource.data.date;
        allow create: if request.auth != null;
      }
    }
  }
}
